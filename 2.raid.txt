# Инициализируем физ. том для использования lvm.
user@Ubuntu:~/Desktop$ sudo pvcreate /dev/sdb1
  Physical volume "/dev/sdb1" successfully created.

user@Ubuntu:~/Desktop$ sudo pvcreate /dev/sdc1
  Physical volume "/dev/sdc1" successfully created.

#Создаем группу томов с именем vg0.
user@Ubuntu:~/Desktop$ sudo vgcreate vg0 /dev/sdb1 /dev/sdc1
  Volume group "vg0" successfully created

# Смотрим что получилось.

# Создаем RAID0 логический том размером 50% от виртуальной группы (vg0) и называем его lv_raid0.
user@Ubuntu:~/Desktop$ sudo lvcreate -i2 -I64 --type raid0 -l 50%VG -n lv_raid0 vg0
  Rounding size <5.00 GiB (1279 extents) up to stripe boundary size 5.00 GiB (1280 extents).
  Logical volume "lv_raid0" created.

# Размечаем файловую систему
user@Ubuntu:~/Desktop$ sudo mkfs.ext4 /dev/vg0/lv_raid0

user@Ubuntu:~/Desktop$ sudo lvs -a -o name,segtype vg0
  LV                  Type  
  lv_raid0            raid0 
  [lv_raid0_rimage_0] linear
  [lv_raid0_rimage_1] linear
