#Функция по проверке корректности введенного password
politic (){
  regex="^[0-9a-zA-Z$%#_^]{8,}$"
   read -p "Введите password: " password
     if [[ "$password" =~ $regex ]]
   then echo "Введеный password соответствует стандартам безопасности для паролей"
  else
   echo "Введеный password не соответствует стандартам безопасности для паролей"
   fi
   }
