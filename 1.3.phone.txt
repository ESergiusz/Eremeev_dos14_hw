#!/bin/bash
#Функция по проверке корректности введенного российского номера
phone (){
   regex="^\+?7\(?[0-9]{3}\)?[0-9]{3}-?[0-9]{2}-?[0-9]{2}$"
   read -p "Введите number: " number
     if [[ "$number" =~ $regex ]]; then
       echo "Введеный number соответствует стандартам формата для российского номера телефона"
     else
       echo "Введеный number не соответствует стандартам формата для российского номера телефона"
   fi
   }
