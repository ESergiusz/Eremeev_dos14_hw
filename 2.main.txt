#!/bin/bash
. ./email.sh
. ./url.sh
. ./phone.sh
. ./politic.sh

PS3="Выберите один из предложенных вариантов: "
select character in "Проверка email" "Проверка адреса" "Проверка номера телефона" "Проверка пароля" "Выход"; do
case $character in
"Проверка email") email  ;;
"Проверка адреса") url ;;
"Проверка номера телефона") phone ;;
"Проверка пароля") politic ;;
"Выход") break;;
*) echo "Может все таки выберите из предложенных варантов?"
esac
done
