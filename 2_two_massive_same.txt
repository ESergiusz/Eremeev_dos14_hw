#!/bin/bash
declare -a massive1; echo "Input Massive #1:"; read -a massive1;
declare -a massive2; echo "Input Massive #2:"; read -a massive2;
declare -a massive3;
for i in ${massive1[@]}; do
  for j in ${massive2[@]}; do
  if [ "$i" == "$j" ]
   then
     massive3+=("$i")
    fi
    done
done
echo "Same elements in two massives:" ${massive3[@]} | uniq
